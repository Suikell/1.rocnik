397  private void form_keyboard(object sender, KeyPressEventArgs e)
398  {
399  // Podmienka pre výpis jediného čísla po stlačení klávesy
400  if (screen.Focused == true)
401  return;
402 
403  // Podmienky pre ošetrenie hodnôt z klávesnice
404  if (e.KeyChar == '=')
405  btn_equal(sender, e);
406  else if (e.KeyChar == 8) // backspace je reprezentovaný ako 8
407  btn_DEL(sender, e);
408  else if (e.KeyChar >= '0' && e.KeyChar <= '9')
409  {
410  test_result();
411  screen.Text += e.KeyChar;
412  }
413  else
414  {
415  // Volanie rozličných metód na základe vybraného vstupu z klávesnice
416  switch (e.KeyChar)
417  {
418  case '+': btn_plus(this, new EventArgs()); break;
419  case '-': btn_minus(this, new EventArgs()); break;
420  case '/': btn_divide(this, new EventArgs()); break;
421  case '*': btn_multiply(this, new EventArgs()); break;
422  case '^': btn_power(this, new EventArgs()); break;
423  case '!': btn_factorial(this, new EventArgs()); break;
424  case '√': btn_sqrt(this, new EventArgs()); break;
425  case '.': btn_decimal_point(this, new EventArgs()); break;
426  case 'l': btn_log(this, new EventArgs()); break;
427  default: MessageBox.Show("Stlačená neplatná klávesa."); break;
428  }
429  }
430  }
