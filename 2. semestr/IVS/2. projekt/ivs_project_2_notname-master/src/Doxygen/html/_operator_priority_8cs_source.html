28  public static double Priority(string input, ref int error_code)
29  {
30  int left, right;
31  double leftOp, rightOp;
32  // definícia poradia priority operatorov
33  char[] op1 = { '!' };
34  char[] op2 = { '*', '/' };
35  char[] op3 = { '^', '√', 'l' };
36  char[] op4 = { '+', '-' };
37  char[][] signPool = { op1, op2, op3, op4 };
38  string[] minusCompinations = { "+-", "*-", "/-" };
39  string calculated;
40  input = input.Replace("log", "l");
41  // hľadanie operátorov z jednotlivých skupín operátorov
42  for (int j = 0; j < 4; j++)
43  {
44  // prechod stringom a vyhľadávanie operátorov z danej skupiny
45  for (int i = input.Substring(1).IndexOfAny(signPool[j]); i != -1; i = input.Substring(1).IndexOfAny(signPool[j]))
46  {
47  i++;
48  char currentSign = input[i]; // aktuálny operátor
49  for (left = i - 1; left > -1 && (char.IsDigit(input[left]) || input[left] == '.' || Array.Exists(minusCompinations, element => element == input.Substring(left, 2)) || (input[left] == '-' && left == 0)); left--) { };
50  left++; // zistenie dĺžky ľavého operátora
51  for (right = i + 1; right < input.Length && (char.IsDigit(input[right]) || input[right] == '.' || input[i + 1] == '-'); right++) { };
52  right--; // zistenie dĺžky pravého operátora
53  leftOp = i - left != 0 ? Convert.ToDouble(input.Substring(left, i - left)) : 0; // nastavenie ľavého operátora
54  rightOp = currentSign != '!' ? right - i != 0 ? Convert.ToDouble(input.Substring(i + 1, right - i)) : 0 : 0;
55  // nastavenie pravého operátora pre binárne operácie
56  switch (currentSign) //riešenie operátora
57  {
58  case '*':
59  calculated = Convert.ToString(MathematicLibrary.Multiply(leftOp, rightOp));
60  break;
61  case '/':
62  calculated = Convert.ToString(MathematicLibrary.Divide(leftOp, rightOp, ref error_code));
63  break;
64  case '+':
65  calculated = Convert.ToString(MathematicLibrary.Add(leftOp, rightOp));
66  break;
67  case '-':
68  calculated = Convert.ToString(MathematicLibrary.Sub(leftOp, rightOp));
69  break;
70  case '!':
71  calculated = Convert.ToString(MathematicLibrary.GetFactorial(leftOp, ref error_code));
72  break;
73  case '^':
74  calculated = Convert.ToString(MathematicLibrary.Pow(leftOp, rightOp, ref error_code));
75  break;
76  case '√':
77  calculated = Convert.ToString(MathematicLibrary.Sqrt(rightOp, leftOp, ref error_code));
78  break;
79  case 'l':
80  calculated = Convert.ToString(MathematicLibrary.Logarithm(rightOp, leftOp, ref error_code));
81  break;
82  default:
83  return (error_code = 7);
84  }
85  if (error_code != 0) //ukončenie kalkulácie v prípade chyby
86  return 1;
87  // nahradenie operátora vypočítanou hodnotou
88  input = input.Replace(input.Substring(left, right - left + 1), calculated);
89  }
90  }
91  // návrat výslednej hodnoty
92  return Convert.ToDouble(input);
93  }
94  }
95 }
